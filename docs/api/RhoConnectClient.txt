# Error codes:
# ERR_NONE = 0
# ERR_NETWORK = 1
# ERR_REMOTESERVER = 2
# ERR_RUNTIME = 3
# ERR_UNEXPECTEDSERVERRESPONSE = 4
# ERR_DIFFDOMAINSINSYNCSRC = 5
# ERR_NOSERVERRESPONSE = 6
# ERR_CLIENTISNOTLOGGEDIN = 7
# ERR_CUSTOMSYNCSERVER = 8
# ERR_UNATHORIZED = 9

def login_callback
  error_code = @params['error_code'].to_i
  if error_code == Rho::RhoError::ERR_NONE
    # run sync if we were successful
    WebView.navigate Rho::RhoConfig.options_path
    RhoConnectClient.doSync
  else
    if error_code == Rho::RhoError::ERR_CUSTOMSYNCSERVER
      @msg = @params['error_message']
    end

    if not @msg or @msg.length == 0
      @msg = Rho::RhoError.new(error_code).message
    end

    WebView.navigate(
      url_for(:action =&gt; :login, :query =&gt; {:msg =&gt; @msg})
    )
  end
end

def search
  page = @params['page'] || 0
  page_size = @params['page_size'] || 10
  RhoConnectClient.search(
    {
      'Contact',
      :from =&gt; 'search',
      :search_params =&gt; {
        :FirstName =&gt; @params['FirstName'],
        :LastName =&gt; @params['LastName'],
        :Company =&gt; @params['Company']
      },
      :offset =&gt; page * page_size,
      :max_results =&gt; page_size
    }
    url_for(:action =&gt; :search_callback)
  )
  render :action =&gt; :search_wait
end

The `search_params` will be forwarded to your callback so you can update the view with the results:<ul class='nav nav-tabs' id='exI4-S1Tab'><li class='active'><a href='#exI4-S1RUBY' data-toggle='tab'>Ruby</a></li></ul><div class='tab-content'><div class='tab-pane active' id='exI4-S1RUBY'><pre class='CodeRay'><code>:::ruby


def search_callback
  status = @params["status"]
  if (status and status == "ok")
  WebView.navigate(
    url_for(
      :action =&gt; :show_page,
      :query =&gt; @params['search_params']
    )
  )
  else
    render :action =&gt; :search_error
  end
end

def show_page
  @contacts = Contact.find(
    :all,
    :conditions =&gt; {
    {
      :func =&gt; 'LOWER',
      :name =&gt; 'FirstName',
      :op =&gt; 'LIKE'
    } =&gt; @params[:FirstName],
    {
      :func =&gt; 'LOWER',
      :name=&gt;'LastName',
      :op=&gt;'LIKE'
    } =&gt; @params[:LastName],
    {
      :func=&gt;'LOWER',
      :name=&gt;'Company',
      :op=&gt;'LIKE'
    } =&gt; @params[:Company],
    },
    :op =&gt; 'OR',
    :select =&gt; ['FirstName','LastName', 'Company'],
    :per_page =&gt; page_size,
    :offset =&gt; page * page_size
  )
  render :action =&gt; :show_page
end

<a name='Remarks'></a>
<h2><i class='icon-warning-sign'></i>Remarks</h2>

<a name='r0'></a><div class='accordion property' id='r0'><div class="accordion-group"><div class="accordion-heading"><span class="accordion-toggle" data-toggle="collapse"  href="#cRemark0"><strong>Notification Parameters</strong><i class="icon-chevron-down pull-left"></i></span></div><div id="cRemark0" class="accordion-body collapse in">  <div class="accordion-inner"><p>When the notification is called, it will receive a variable called @params, just like a normal Rhodes controller action.</p>

<h2>Common Parameters</h2>

<p>These parameters are included in all notifications.</p>

<ul>
<li>source_id &ndash; The id of the current model that is synchronizing.</li>
<li>source_name &ndash; Name of the model (i.e. &ldquo;Product&rdquo;)</li>
<li>sync_type &ndash; Type of sync used for this model: &ldquo;incremental&rdquo; or &ldquo;bulk&rdquo;</li>
<li>status &ndash; Status of the current sync process. See below for the possible values:</li>
</ul>


<h3>in_progress &ndash; incremental sync progress</h3>

<ul>
<li>@params[&ldquo;total_count&rdquo;] &ndash; Total number of records that exist for this RhoConnect source.</li>
<li>@params[&ldquo;processed_count&rdquo;] &ndash; Number of records included in the sync page.</li>
<li>@params[&ldquo;cumulative_count&rdquo;] &ndash; Number of records the SyncEngine has processed so far for this source.</li>
</ul>


<h3>in_progress &ndash; bulk sync progress</h3>

<ul>
<li><p>@params[&ldquo;bulk_status&rdquo;] &ndash; The state of the bulk sync process:</p>

<ul>
<li>&ldquo;start&rdquo; &ndash; Bulk sync has started for a specific partition</li>
<li>&ldquo;download&rdquo; &ndash; Bulk sync file download has started</li>
<li>&ldquo;change_db&rdquo; &ndash; New bulk sync database change has started</li>
<li>&ldquo;blobs&rdquo; &ndash; Bulk sync blob files have started to download</li>
<li>&ldquo;ok&rdquo; &ndash; Current partition has completed</li>
<li>&ldquo;complete&rdquo; &ndash; All partitions have completed</li>
</ul>
</li>
<li><p>@params[&ldquo;partition&rdquo;] &ndash; Current bulk sync partition.</p></li>
</ul>


<h3>error</h3>

<ul>
<li>@params[&ldquo;error_code&rdquo;] &ndash; HTTP response code of the RhoConnect server error: 401, 500, 404, etc.</li>
<li>@params[&ldquo;error_message&rdquo;] &ndash; Response body (if any)</li>
<li>@params[&ldquo;server_errors&rdquo;] &ndash; Hash of Type objects of RhoConnect adapter error (if exists):

<ul>
<li>&ldquo;login-error&rdquo; &ndash; An error in adapter login method</li>
<li>&ldquo;query-error&rdquo; &ndash; An error in adapter query method</li>
<li>&ldquo;create-error&rdquo; &ndash; An error in adapter create method</li>
<li>&ldquo;update-error&rdquo; &ndash; An error in adapter update method</li>
<li>&ldquo;delete-error&rdquo; &ndash; An error in adapter delete method</li>
<li>&ldquo;logoff-error&rdquo; &ndash; An error in adapter logoff method</li>
</ul>
</li>
</ul>


<p>Each error contains a &lsquo;message&rsquo; key with the error message.  For example:</p>

<pre><code>:::ruby
@params["server_errors"]["query-error"]["message"] 
#=&gt; "Error connecting to backend server: http://rhostore.herokuapp.com"
</code></pre>

<h4>Handling &lsquo;create-error&rsquo;</h4>

<p>&ldquo;create-error&rdquo; is a unique situation that needs to be handled in your sync callback.  See <code>RhoConnectClient.onSyncCreateError()</code> for more details.</p>

<h3>complete</h3>

<p>This status returns only when the sync run is complete (all partitions are done synchronizing).</p>

<h3>schema-changed</h3>

<p>This is present if a FixedSchema model has changed during the current sync run.  A new bulk sync run should be triggered (by setting bulksyncState) and the user should be alerted in the UI.</p>
